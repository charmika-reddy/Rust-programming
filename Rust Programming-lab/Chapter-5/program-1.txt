//Given a vector with an even number of elements, remove the last element from the input
vector, and then the middle element. Then insert the sum of the remaining elements to the
end of the resulting vector.

fn main() {
    let mut v = vec![2, 4, 6, 8, 10, 12]; // even length

    // Remove last element
    v.pop();

    // Remove middle element
    let mid = v.len() / 2;
    v.remove(mid);

    // Insert sum at the end
    let sum: i32 = v.iter().sum();
    v.push(sum);

    println!("Resulting vector: {:?}", v);
}
