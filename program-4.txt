//. Write a function calculate_area_perimeter() that takes x and y( length and width of a 
rectangle) as a parameter to the function and returns a tuple (area, perimeter). 

fn calculate_area_perimeter(x: i32, y: i32) -> (i32, i32) {
    let area = x * y;
    let perimeter = 2 * (x + y);
    (area, perimeter)
}

fn main() {
    let length = 5;
    let width = 3;
    let (area, perimeter) = calculate_area_perimeter(length, width);

    println!("Area: {}", area);
    println!("Perimeter: {}", perimeter);
}
